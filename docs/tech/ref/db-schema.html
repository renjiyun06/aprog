完整 DDL

-- internal/controlplane/db/migrations/0001_init.sql
-- +goose Up

CREATE TABLE users (
    id              TEXT PRIMARY KEY,           -- nanoid 16
    username        TEXT NOT NULL UNIQUE COLLATE NOCASE,
    password_hash   TEXT NOT NULL,              -- argon2id encoded string
    created_at      INTEGER NOT NULL,           -- unix seconds
    last_login_at   INTEGER
);

CREATE TABLE auth_sessions (
    id              TEXT PRIMARY KEY,           -- = token, 32 random bytes base64url
    user_id         TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    created_at      INTEGER NOT NULL,
    last_used_at    INTEGER NOT NULL,
    expires_at      INTEGER NOT NULL,           -- = created_at + 30d default
    revoked_at      INTEGER
);

CREATE INDEX idx_auth_sessions_user ON auth_sessions(user_id);
CREATE INDEX idx_auth_sessions_active ON auth_sessions(expires_at)
    WHERE revoked_at IS NULL;

CREATE TABLE processes (
    pid                 TEXT PRIMARY KEY,       -- yyMMdd-<4 nanoid>
    program             TEXT NOT NULL,
    user_id             TEXT NOT NULL REFERENCES users(id),
    state               TEXT NOT NULL CHECK(state IN ('spawned','running','exited')),
    phase               TEXT,
    snapshot_path       TEXT NOT NULL UNIQUE,
    sandbox_id          TEXT,                   -- nullable
    sandbox_started_at  INTEGER,
    created_at          INTEGER NOT NULL,
    last_attached_at    INTEGER,
    last_exited_at      INTEGER
);

CREATE INDEX idx_processes_user_state ON processes(user_id, state);
CREATE INDEX idx_processes_sandbox ON processes(sandbox_id) WHERE sandbox_id IS NOT NULL;

-- +goose Down
DROP TABLE processes;
DROP TABLE auth_sessions;
DROP TABLE users;

字段细节

字段决策为什么
users.username COLLATE NOCASE大小写无关唯一避免 Lamarck 跟 lamarck 同时存在
auth_sessions.id = token不哈希token 本身就是 256-bit 随机串, 等价于 hash. 减少一层 lookup
idx_auth_sessions_activepartial index只有未 revoke 的入索引, 减小, 校验时直接命中
timestamps INTEGERunix 秒SQLite 没原生 TIMESTAMP. sqlc 配 emit_type_overrides 转 time.Time
state CHECK constraint枚举防止意外写入 'killed' 之类不存在的状态
processes.snapshot_path UNIQUE1:1 反向唯一不可能两个 process 共用一个快照路径
idx_processes_sandboxpartial反查 sandbox_id → process 很常见 (Daytona webhook 等), 但只有 running 的进程有 sandbox_id
ON DELETE CASCADE on auth_sessions跟随 user 删用户删除时 sessions 一并清掉 (MVP 不开放 delete, 但语义先对)
processes.user_id 不 CASCADE禁止删 user 如果有 process状态永不删除, user 也不能孤立

argon2id 参数

// internal/controlplane/auth/hash.go
import "golang.org/x/crypto/argon2"

// 2026 OWASP 推荐: t=1, m=46MB, p=1 起步. aprog 用 t=2, m=64MB, p=4 给 server-grade.
const (
    argonTime    uint32 = 2
    argonMemory  uint32 = 64 * 1024       // 64 MiB
    argonThreads uint8  = 4
    argonKeyLen  uint32 = 32
    argonSaltLen        = 16
)

func hashPassword(pw string) (string, error) {
    salt := make([]byte, argonSaltLen)
    if _, err := rand.Read(salt); err != nil { return "", err }
    h := argon2.IDKey([]byte(pw), salt, argonTime, argonMemory, argonThreads, argonKeyLen)
    return fmt.Sprintf("$argon2id$v=19$m=%d,t=%d,p=%d$%s$%s",
        argonMemory, argonTime, argonThreads,
        base64.RawStdEncoding.EncodeToString(salt),
        base64.RawStdEncoding.EncodeToString(h)), nil
}

func verifyPassword(encoded, pw string) bool {
    // parse $argon2id$v=19$m=...,t=...,p=...$salt$hash
    parts := strings.Split(encoded, "$")
    if len(parts) != 6 || parts[1] != "argon2id" { return false }
    var memory, time uint32; var threads uint8
    fmt.Sscanf(parts[3], "m=%d,t=%d,p=%d", &memory, &time, &threads)
    salt, _ := base64.RawStdEncoding.DecodeString(parts[4])
    want, _ := base64.RawStdEncoding.DecodeString(parts[5])
    got := argon2.IDKey([]byte(pw), salt, time, memory, threads, uint32(len(want)))
    return subtle.ConstantTimeCompare(got, want) == 1
}

sqlc 用法

查询文件 internal/controlplane/db/queries/*.sql:

-- queries/users.sql
-- name: GetUserByUsername :one
SELECT * FROM users WHERE username = ?;

-- name: CreateUser :one
INSERT INTO users (id, username, password_hash, created_at)
VALUES (?, ?, ?, ?)
RETURNING *;

-- name: TouchUserLogin :exec
UPDATE users SET last_login_at = ? WHERE id = ?;


-- queries/auth_sessions.sql
-- name: CreateAuthSession :one
INSERT INTO auth_sessions (id, user_id, created_at, last_used_at, expires_at)
VALUES (?, ?, ?, ?, ?) RETURNING *;

-- name: GetActiveSession :one
SELECT * FROM auth_sessions
 WHERE id = ?
   AND revoked_at IS NULL
   AND expires_at > unixepoch();

-- name: TouchSession :exec
UPDATE auth_sessions SET last_used_at = unixepoch() WHERE id = ?;

-- name: RevokeSession :exec
UPDATE auth_sessions SET revoked_at = unixepoch() WHERE id = ?;


-- queries/processes.sql
-- name: CreateProcess :one
INSERT INTO processes (pid, program, user_id, state, snapshot_path, created_at)
VALUES (?, ?, ?, 'spawned', ?, ?) RETURNING *;

-- name: GetProcessForUser :one
SELECT * FROM processes WHERE pid = ? AND user_id = ?;

-- name: ListUserProcesses :many
SELECT * FROM processes
 WHERE user_id = ?
   AND (sqlc.arg('include_exited')::int = 1 OR state != 'exited')
 ORDER BY created_at DESC;

-- name: MarkRunning :exec
UPDATE processes
   SET state = 'running', sandbox_id = ?, sandbox_started_at = ?, last_attached_at = ?
 WHERE pid = ?;

-- name: MarkExited :exec
UPDATE processes
   SET state = 'exited', sandbox_id = NULL, last_exited_at = ?
 WHERE pid = ?;

-- name: TouchAttached :exec
UPDATE processes SET last_attached_at = ? WHERE pid = ?;

未决问题

  • Postgres 迁移路径: sqlc 支持 PG. DDL 大体兼容, 但 COLLATE NOCASE 要换成 citext, unixepoch() 换 EXTRACT(EPOCH FROM NOW()), partial index 语法略调. MVP 不动.
  • sandbox_history 表: 现状只留当前 sandbox_id, 丢失历史. 真要做审计 / compute 账单, 加 sandbox_history(pid, sandbox_id, started_at, ended_at, ...) 表. 已在 data model 提到.
